\section{Proof of Proposition \ref{prop:strat}}\label{sec: Proof of Proposition 1}\label{sec:app:prop_strat}
\begin{proof}
    Fix a CGMRP 
    $\M = (\P, \V)$  with underlying CGFRP $\P = (\F, \preceq)$. Assume strategy $\js_\Coalition$ of coalition $\Coalition$ is strongly beneficial for group $\GroupB\subseteq \AGT$ at state $s$. Then $\outset_\F(s,\js_\Coalition)\subseteq \Best_{i,s}$ for every $i\in \GroupB $. Let $i\in\GroupB$, $\history\in \outset_\F(s,\js_\Coalition)$ and $\history'\in \Comp_{\F,s}-\outset_\F(s,\js_\Coalition)$. Then $\history\in \Best_{i,s}$. Therefore, $\history' \preceq_{i,s} \history$. Then $\js_\Coalition\in \JsWB_{\M }^C(s,\GroupB)$.

    The proof for harmful strategies is similar.
  \end{proof}
  
\section{Proof of completeness for $\ourlogic$}\label{sec: proof of completeness for SCA-CL}\label{sec:app:completeness}
In this section, we show the proof of completeness for $\ourlogic$ in details. The key step is constructing a model for any satisfiable standard conjunction, which equals to the negation of a standard disjunction. We use a method called `blueprint' to achieve it.

For convenience, we use some notations in this section. For a structure $S$, we use $S.component$ to denote the component of $S$. For a tuple $t$, we use $t.i$ to denote the $i$-th element of $t$. 

\subsection{Some derivabilities}\label{section: some derivabilities}
\begin{theorem}
  The following are derivable in $\ourlogic$:
  %\tag{$\mathtt{A}\text{-}\mathtt{MC}1$} \label{ax:MC1}
  \begin{align*}  
    %No absurd rational actions
    & \dfrac{\; \phi \;}{\; \atlbh{\Coalition}{B}{H}\G^\ns \phi \;} 
  \tag{$\mathtt{Gen}_{\G^\ns}$} \label{ir:Gen_Gw}
  \\
    & 
     \atlop{\emptyset}\G^\ns\big (\theta \rightarrow \atlbh{\Coalition}{B}{H}\X(\phi \wedge \theta)\big ) \\
     &
     \rightarrow \atlop{\emptyset }\G^\ns (\theta \rightarrow \atlbh{\Coalition}{B}{H}\G \phi)  \tag{$\mathtt{th}\text{-}\mathtt{GFP}_\G$} \label{th:GFP_G}
     \\
    & 
     \atlop{\emptyset}\G^\ns\big (\theta \rightarrow (\phi \wedge \atlbh{\Coalition}{B}{H}\X\theta)\big ) \\
     &
     \rightarrow \atlop{\emptyset }\G^\ns (\theta \rightarrow \atlbh{\Coalition}{B}{H}\G^\ns \phi)  \tag{$\mathtt{th}\text{-}\mathtt{GFP}_{\G^\ns}$} \label{th:GFP_Gw}
     \\
    & 
     \dfrac{\; \theta \to \atlbh{\Coalition}{B}{H}\X(\phi \wedge \theta) \;}{\; \theta \to \atlbh{\Coalition}{B}{H}\G \phi\;}  \tag{$\mathtt{In}_\G$} \label{ir:In_G}
  \end{align*}
  \end{theorem}  


\subsection{Canonical realization}\label{sec: canonical realization}
In this section, for any canonical state $s\in \St^\psi$ and formula $\varphi$, we write $s\vdash \varphi $ if $s.1\vdash \varphi $, and denote $| \varphi | = \{s\in \St^\psi \mid s\vdash \varphi \}$.

\begin{definition}[Characteristic structure]\label{def: Characteristic
structure}
  Let $\Omega \subseteq \St^\psi$ and $(\chi ,\mathfrak{B},\mathfrak{H}) \in \ecl(\psi)\times \powerset{\powerset{\AGT}}\times \powerset{\powerset{\AGT}}$, we say $(\chi ,\mathfrak{B},\mathfrak{H})$ is a characteristic structure of $\Omega$ iff for all $(X ,B,H)\in \St^\psi$ we have $(X ,B,H)\in \Omega$ iff $X\vdash \chi $, $B \in \mathfrak{B}$ and $H \in \mathfrak{H}$.
\end{definition}

\begin{theorem}
    Every $\Omega \subseteq \St^\psi$ has a characteristic structure.
\end{theorem}
\begin{proof} 
  Let $\Omega \subseteq \St^\psi$. 
  Let $\chi = \bigvee_{s\in \Omega}\bigwedge_{\varphi\in s.1}\varphi$, $\mathfrak{B} = \{B\subseteq \AGT\mid B=s.2 \text{ for some }s\in \Omega\}$, $\mathfrak{H} = \{H\subseteq \AGT\mid H=s.3 \text{ for some }s\in \Omega\}$. Note $\chi $ is a boolean combination of formulas in $\cl(\psi)$, then there is $\chi' \in \ecl(\psi)$ such that $\vdash \chi \leftrightarrow \chi'$. Then $(\chi',\mathfrak{B},\mathfrak{H})$ is a characteristic structure of $\Omega$.
\end{proof}

\begin{lemma}[Existence lemma]
For all canonical state $s\in \St^\psi$ and $\atlbh{\Coalition}{B}{H}\X \chi\in \ecl(\psi)$, if $s.1\nvdash \atlbh{\Coalition}{B}{H}\X \chi$, then for every available $\Coalition$-action $\ja_C$ there is a state $s'\in \R_{\ja_C}(s)$ such that $s'.1\vdash \neg \chi $, $B\nsubseteq s.2$, or $H\nsubseteq s.3$.
\end{lemma}
\begin{proof}
Fix a canonical state $s\in \St^\psi$ and $\atlbh{\Coalition}{B}{H}\X \chi\in \ecl(\psi)$ with $s.1\nvdash \atlbh{\Coalition}{B}{H}\X \chi$.
Consider whether $C=\AGT $.
\begin{itemize}
  \item Assume $C=\AGT $. Then by the definition of candidate successor, for all $s'\in \CS(s,\ja_C)$ we have $s'.1\vdash \neg \chi $, $B\nsubseteq s'.2$, or $H\nsubseteq s'.3$. Therefore, there is a state $s'\in \R_{\ja_C}(s)$ such that $s'.1\vdash \neg \chi $, $B\nsubseteq s'.2$, or $H\nsubseteq s'.3$.
  %
  \item Assume $C\neq \AGT $. Then there is $a\in \AGT\setminus \Coalition $. Therefore, there is $\ja_\AGT:\AGT\to \ACT^\psi $ such that $\ja_\Coalition \subseteq \ja_\AGT $, $\te(\ja_\AGT)=a$ and $\big((\ja_\AGT(a).3)(\CS(s,\ja_C))\big).1\vdash \neg \chi $, $B\nsubseteq \big((\ja_\AGT(a).3)(\CS(s,\ja_C))\big).2$, or $H\nsubseteq \big((\ja_\AGT(a).3)(\CS(s,\ja_C))\big).3$.
    Therefore, there is a state $s'\in \R_{\ja_C}(s)$ such that $s'.1\vdash \neg \chi $, $B\nsubseteq s'.2$, or $H\nsubseteq s'.3$.
\end{itemize}
\end{proof}

\begin{theorem}[Local realization lemma]
For every $\Coalition,B,H\in\powerset{\AGT}$, 
canonical state $s\in \St^\psi$, 
and $\chi \in \ecl(\psi )$, 
we have: $s.1\vdash \atlbh{\Coalition}{B}{H}\X\chi $ iff 
there is a joint action $\ja_C$ of $\Coalition$ such that 
$\js_\Coalition $ is available at $s$ and 
for all $s'\in \R_{\ja_C}(s)$ we have 
$s'.1\vdash \chi $, $B\subseteq s'.2$ and $H\subseteq s'.3$.
\end{theorem}
\begin{proof}
    Fix coaltition $\Coalition,B,H\in\powerset{\AGT}$, 
canonical state $s\in \St^\psi$, 
and $\chi \in \ecl(\psi )$.  From left to right, assume $s.1\vdash \atlbh{\Coalition}{B}{H}\X\chi $. Let $\ja_\Coalition: \Coalition \to \ACT^\psi$ such that $\Fsupport(\ja_\Coalition(a))=\atlbh{\Coalition}{B}{H}\X\chi $. Then $\atlbh{\Coalition}{B}{H}\X\chi \in\Consesus(\ja_\AGT)$ for all $\ja_\AGT:\AGT\to \ACT^\psi$ with $\ja_\Coalition \subseteq \ja_\AGT $. Note there is $\ja_\AGT:\AGT\to \ACT^\psi $ such that $\ja_\Coalition \subseteq \ja_\AGT $, $\Fsupport(\ja_\AGT (b))= \atlop{\emptyset}\X\top$ for all $b\in\AGT-\Coalition$ and $\ja_\AGT $ is available at $s$.    
    Then $\ja_\Coalition$ is available at $s$. Let $\ja_\AGT:\AGT\to \ACT^\psi$ with $\ja_\Coalition \subseteq \ja_\AGT $. Note $\atlbh{\Coalition}{B}{H}\X\chi \in\Consesus(\ja_\AGT)$ and $s.1\vdash \atlbh{\Coalition}{B}{H}\X\chi $. Then $s'.1\vdash \chi $, $B\subseteq s'.2$ and $H\subseteq s'.3$ for all $s'\in \CS(s,\ja_\AGT )$.

    The deriction from right to left follows from existence lemma.
  \end{proof}

\begin{theorem}[$\G$-realization lemma]
For $s\in \St^\psi $ and $\atlbh{\Coalition}{B}{H}\G \phi \in ecl(\phi)$, we have $\atlbh{\Coalition}{B}{H}\G \phi\in s$ iff there is a $\Coalition$-strategy $\js_\Coalition $ such that for all $\history \in \outset(s,\js_\Coalition)$ and all $i>0$, we have $\phi \in \history(i).1$ and $B\subseteq \history(i).2$ and $H\subseteq \history(i).3$.
\end{theorem}
\begin{proof}
 Fix $s\in \St^\psi $ and $\atlbh{\Coalition}{B}{H}\G \phi \in ecl(\phi)$. From left to right, suppose that $\atlbh{\Coalition}{B}{H}\G \phi\in s$. By axiom \ref{ax:FP_G} and Local Realization lemma, for all $s'\in |\atlbh{\Coalition}{B}{H}\G \phi |$, there is a joint action $\ja_{\Coalition}$ such that $\R_{\ja_{\Coalition}}(s')\subseteq |\phi \wedge \atlbh{\Coalition}{B}{H}\G \phi |$. Let $\js_\Coalition $ be a $\Coalition$-strategy $\js_\C$ such that for all $s'\in |\atlbh{\Coalition}{B}{H}\G \phi |$ we have $\js_\C(s)$ is a joint action $\ja_{\Coalition}$ such that $\R_{\ja_{\Coalition}}(s')\subseteq |\phi \wedge \atlbh{\Coalition}{B}{H}\G \phi |$, $B\subseteq s''.2$ and $H\subseteq s''.3$. Let $\history \in \outset(s,\js_\Coalition)$, by induction on $i>0$, we can show that $\phi \in \history(i).1$, $B\subseteq \history(i).2$ and $H\subseteq \history(i).3$. 

 From right to left, suppose that there is a $\Coalition$-strategy $\js_\Coalition $ such that $\phi \in \history(i).1$, $B\subseteq \history(i).2$ and $H\subseteq \history(i).3$ for all $\history \in \outset(s,\js_\Coalition)$ and all $i>0$. 
 Let $\Omega = \{s\}\cup \{\history[i]\mid \history \in \outset(s,\js_\Coalition), i>0\}$ and $(\chi ,\mathfrak{B},\mathfrak{H})$ be the characteristic structure of $\Omega$. 
 Note $\R_{\js_\Coalition(s')}(s')\subseteq \Omega$ for all $s'\in \Omega $. Then $s'\vdash \atlbh{\Coalition}{B}{H}\X (\phi \wedge \chi )$ for all $s'\in \Omega $. 
 Note $\chi \in s.1$. It suffices to show that $\vdash \chi \rightarrow \atlbh{\Coalition}{B}{H}\G \phi $. 
 By \ref{ir:In_G}, 
 it suffices to show that $\vdash \chi \rightarrow \atlbh{\Coalition}{B}{H}\X(\phi \wedge \chi)$. 
 Let $s^+$ be a maximal consistent set such that $\chi \in s^+$. Note $s^+\cap ecl(\phi )\in \St^\psi $. Then $s^+\cap ecl(\phi )\in \Omega$. 
 Therefore, $s^+\cap ecl(\phi )\vdash \atlbh{\Coalition}{B}{H}\X (\phi \wedge \chi )$. Therefore, $\vdash \chi \rightarrow \atlbh{\Coalition}{B}{H}\X(\phi \wedge \chi)$. Therefore, $\vdash \chi \rightarrow \atlbh{\Coalition}{B}{H}\G \phi $ and $\atlbh{\Coalition}{B}{H}\G \phi\in s.1$.
\end{proof}

\section{Proof of  Theorem \ref{theorem: tree-like property} }\label{sec: proof of tree-like property}\label{sec:app:tree_like}
In this section, we show the proof of tree-like property with unfolding. 
\begin{proof}
Let $\varphi \in \lang_{\atlBDlogic} $. From left to right, suppose $\varphi $ is satisfiable on pointed CGSRP $(\M,s)$ where $\M=(\St,  \ACT,  \R, \preceq, \V)$ and $s\in \St$. We construct a new model $\M'=(\St',  \ACT',  \R', \preceq', \V')$ as the following:
\begin{itemize}
  \item $\St' = \PPath_{\M}$ is the set of all partial path in $\M$;
  \item $\ACT' = \ACT$;
  \item $\R'_\ja = \{(\history, \history')\in \St'\mid \last(\history)\R_\ja \last(\history')\}$;
  \item $\preceq'_{a,\history} = \{(\history',\history'')\in \Comp_{\M,,\history}\mid \last(\history')\preceq_{a,\last(\history)} \last(\history'')\}$;
  \item $\V'(\history) = \V(\last(\history))$.
  \end{itemize}
  Note every strategy $\js_\Coalition$ in $\M$ corresponds to a strategy $\js'_\Coalition$ in $\M'$ with $\js'_\Coalition(\history) = \js_\Coalition(\last(\history))$ for all $\history \in \PPath_{\M,}$, and for all $\history' \in \outset_{\M'}(\history,\js'_\Coalition)$ we have $\last(\history')\in \outset_{\M}(\last(\history),\js_\Coalition)$. 
  Then by induction on the structure of $\varphi$, we can show that $(\M,s)\models \varphi $ iff $(\M',\langle s\rangle)\models \varphi $, where $\langle s\rangle$ is the singleton path that only contains $s$. 

  The direction from right to left is straightforward as a tree-like CGMRP is a CGMRP.
\end{proof}

\section{Proof of Theorem \ref{theo:embedding} }\label{sec: embeding perserve satifability}\label{sec:app:embedding}
\label{sec: Proof for embedding}

\begin{proof} 
We prove the 
first statement of the theorem. 
Suppose $\varphi \in \lang_{\atlBDlogic} $
is satisfiable.
Thus, by Theorem \ref{theorem: tree-like property},
there exists a CGSRP
$\M=(\St,  \ACT,  \R, \preceq, \V)$ 
  and $s_0 \in \St $
  such that $(\M,s_0 )\models \varphi $. 
  Define a new structure
  $\M'=(\St,  \ACT,  \R, \preceq, \V')$
 such that for all $s\in \St$
  \begin{align*}
    \V'(s)= &\V(s)\cup \{\help_a\mid \exists r\in \R^-(s)
     \exists \history \in \Best_{a,s}\text{ s.t. } s= \history(1)\}\\
     &\cup \{\harm_a\mid \exists r\in \R^-(s)
     \exists \history \in \Worst_{a,s}\text{ s.t. } s= \history(1)\}
  \end{align*}
Clearly, $\M' $ is a CGSRP
and the only difference between $\M'$
and $\M$
is in the valuation function 
that does not affect the relational properties
of the structure. 
By induction on the structure of the formula,
it is routine to prove that 
``$(\M,v )\models \mathit{tr}( \varphi )$ iff
  $(\M',v )\models \varphi $'' for all $v \in W$.
  The other direction is proved in an analogous way. 

The proof of
the second statement is based 
on a straightforward adaptation of the above construction without requiring stable preferences.
\end{proof}
